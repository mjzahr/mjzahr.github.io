@inproceedings{washabaugh2012nonlinear,
  abstract = {A model reduction framework based on the concept of local reduced-order bases is presented. The offline phase of the method builds the local reduced-order bases using an unsupervised learning algorithm. In the online phase of the method, the choice of the local basis is based on the current state of the system. Inexpensive rank-one updates to the local bases are performed during the online phase for increased accuracy. Applications to nonlinear CFD simulations show that the method is effective in producing small and accurate reduced order models.},
