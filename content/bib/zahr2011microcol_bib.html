@techreport{zahr2011microcol,
  abstract = {This work presents a model for the use of an ensemble of microcolumns as angular motion sensors. The methodology for angular detection is most suitable for small, high-frequency systems. This project makes use of the Equipartition Theorem and the Louisville Equation to determine the evolution of the probability density distribution of the location of the microcolumn tips. The work formulates the problem of interest, establishes a theoretical and computational model, presents the numerical stability analysis, and the simulation results. The Two-Step Lax Wendroff method was the explicit integration scheme used.},
